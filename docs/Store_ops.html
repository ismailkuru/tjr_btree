Module Store_ops

module Store_ops: sig .. end
Store operations: alloc, free and read. The store is a level above the raw disk block device, on which we build the B-tree.
type ('k, 'v, 'r, 't) store_ops = {
   store_free : 'r list -> (unit, 't) Monad.m;
   store_read : 'r -> (('k, 'v, 'r) Frame.frame, 't) Monad.m;
   store_alloc : ('k, 'v, 'r) Frame.frame -> ('r, 't) Monad.m;
}

       'e
